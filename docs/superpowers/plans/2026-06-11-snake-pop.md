# Snake Pop Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build "Snake Pop" — a slither-style arcade game where your snake body is a match-3 board — as a standalone PWA in `apps/snake`.

**Architecture:** Canvas 2D game loop at 60fps using `requestAnimationFrame`; all game state is plain TypeScript classes/objects managed in a single `GameEngine` class; React is used only for HUD/overlay DOM elements overlaid on the canvas. Zustand stores settings (theme, reduced-motion, sound) persisted to localStorage via the same `storage.ts` pattern as `apps/web`.

**Tech Stack:** React 18, TypeScript, Vite + vite-plugin-pwa, Zustand, canvas-confetti, Web Audio API, Playwright (Chromium / Pixel 7 only)

```typescript
const PREFIX = 'sp.v1.';

export function load<T>(key: string, fallback: T): T {
  try {
    const raw = localStorage.getItem(PREFIX + key);
    if (raw === null) return fallback;
    return JSON.parse(raw) as T;
  } catch {
    return fallback;
  }
}

export function save(key: string, value: unknown): void {
  try {
    localStorage.setItem(PREFIX + key, JSON.stringify(value));
  } catch {
    // storage full or privacy mode
  }
}
```

```typescript
let ctx: AudioContext | null = null;

function getCtx(): AudioContext {
  if (!ctx) ctx = new AudioContext();
  return ctx;
}

function blip(freq: number, dur: number, vol = 0.15): void {
  try {
    const c = getCtx();
    const osc = c.createOscillator();
    const gain = c.createGain();
    osc.connect(gain);
    gain.connect(c.destination);
    osc.frequency.value = freq;
    osc.type = 'sine';
    gain.gain.setValueAtTime(vol, c.currentTime);
    gain.gain.exponentialRampToValueAtTime(0.001, c.currentTime + dur);
    osc.start(c.currentTime);
    osc.stop(c.currentTime + dur);
  } catch { /* AudioContext blocked */ }
}

export function playEat(): void   { blip(660, 0.08, 0.12); }
export function playPop(): void   { blip(880, 0.12, 0.2); blip(1100, 0.08, 0.15); }
export function playDeath(): void { blip(200, 0.4, 0.2); blip(150, 0.6, 0.15); }
export function playBest(): void  { blip(880, 0.1); blip(1100, 0.1); blip(1320, 0.2); }
```

```typescript
import { create } from 'zustand';
import { load, save } from '../lib/storage';

export type ThemeChoice = 'auto' | 'light' | 'dark';

interface SettingsState {
  theme: ThemeChoice;
  reducedMotion: boolean;
  sound: boolean;
  set: (patch: Partial<Omit<SettingsState, 'set'>>) => void;
}

const stored = load<Partial<SettingsState>>('settings', {});

export const useSettings = create<SettingsState>((set, get) => ({
  theme: stored.theme ?? 'auto',
  reducedMotion: stored.reducedMotion ?? false,
  sound: stored.sound ?? false,
  set: (patch) => {
    set(patch);
    const { theme, reducedMotion, sound } = get();
    save('settings', { theme, reducedMotion, sound });
  },
}));

export function applyTheme(theme: ThemeChoice, reducedMotion: boolean): void {
  const dark = window.matchMedia('(prefers-color-scheme: dark)').matches;
  const resolved = theme === 'auto' ? (dark ? 'dark' : 'light') : theme;
  document.documentElement.dataset.theme = resolved;
  document.documentElement.dataset.motion = reducedMotion ? 'off' : 'on';
  const meta = document.querySelector<HTMLMetaElement>('meta[name="theme-color"]')
    ?? (() => {
      const m = document.createElement('meta');
      m.name = 'theme-color';
      document.head.appendChild(m);
      return m as HTMLMetaElement;
    })();
  meta.content = resolved === 'dark' ? '#0a0a1a' : '#eef2ff';
}
```

- [ ] **Step 1: Create `apps/snake/src/game/engine.ts`**

```typescript
import {
  ARENA_W, ARENA_H, SEG_R, SEG_SPACING,
  SPEED_BASE, SPEED_BOOST, BOOST_CONSUME_RATE, BOOST_MIN_LENGTH,
  COLOR_COUNT, POP_WARN_MS, COMBO_WINDOW_MS,
  SCORE_BASE, RIVAL_COUNT, RIVAL_RESPAWN_MS, RIVAL_SPEED,
  ORB_COUNT, ORB_R, SHAKE_PX, SHAKE_MS, SHAKE_MIN_POP, INITIAL_LENGTH,
} from './constants';
import type { GameState, Segment, Orb, Rival, PendingPop, Particle, Vec2 } from './types';
import { load, save } from '../lib/storage';

function rand(min: number, max: number): number {
  return min + Math.random() * (max - min);
}

function randInt(min: number, max: number): number {
  return Math.floor(rand(min, max + 1));
}

function dist(a: Vec2, b: Vec2): number {
  const dx = a.x - b.x, dy = a.y - b.y;
  return Math.sqrt(dx * dx + dy * dy);
}

function spawnOrb(id: number, exclude?: Vec2[]): Orb {
  let x: number, y: number;
  const PAD = 80;
  do {
    x = rand(PAD, ARENA_W - PAD);
    y = rand(PAD, ARENA_H - PAD);
  } while (exclude && exclude.some(p => dist({ x, y }, p) < SEG_SPACING * 6));
  return { id, x, y, color: randInt(0, COLOR_COUNT - 1), scale: 1 };
}

function makeRival(id: number, now: number): Rival {
  const cx = rand(ARENA_W * 0.1, ARENA_W * 0.9);
  const cy = rand(ARENA_H * 0.1, ARENA_H * 0.9);
  const segs: Segment[] = [];
  for (let i = 0; i < 8; i++) {
    segs.push({ x: cx - i * SEG_SPACING, y: cy, color: randInt(0, COLOR_COUNT - 1) });
  }
  return { id, segments: segs, targetOrb: null, dead: false, respawnAt: now };
}

export function createInitialState(): GameState {
  const best = load<number>('best', 0);
  const cx = ARENA_W / 2, cy = ARENA_H / 2;
  const segments: Segment[] = [];
  for (let i = 0; i < INITIAL_LENGTH; i++) {
    segments.push({ x: cx - i * SEG_SPACING, y: cy, color: i % COLOR_COUNT });
  }
  const now = performance.now();
  const orbs: Orb[] = [];
  for (let i = 0; i < ORB_COUNT; i++) orbs.push(spawnOrb(i, segments));
  const rivals: Rival[] = [];
  for (let i = 0; i < RIVAL_COUNT; i++) rivals.push(makeRival(i, now));

  return {
    phase: 'playing',
    score: 0,
    best,
    combo: 0,
    comboExpiry: 0,
    segments,
    pointer: { x: cx + 60, y: cy },
    pointerDown: false,
    boostConsumeAcc: 0,
    orbs,
    rivals,
    pendingPops: [],
    particles: [],
    shakeUntil: 0,
    nextOrbId: ORB_COUNT,
    headSquash: 1,
    headSquashDecay: 0,
  };
}

export class GameEngine {
  state: GameState;

  constructor(state: GameState) {
    this.state = state;
  }

  // ── public API (called by GameView) ────────────────────────────────────────

  setPointer(worldX: number, worldY: number, down: boolean): void {
    this.state.pointer = { x: worldX, y: worldY };
    this.state.pointerDown = down;
  }

  tick(dt: number, now: number): void {
    if (this.state.phase !== 'playing') return;
    this._movePlayer(dt, now);
    this._checkEat(now);
    this._fireReadyPops(now);
    this._moveRivals(dt, now);
    this._checkRivalCollisions(now);
    this._updateParticles(dt);
    this._updateHeadSquash(dt);
  }

  /** Spawn + instantly eat an orb of color c (testability hook). */
  eatColor(c: number): void {
    if (this.state.phase !== 'playing') return;
    const head = this.state.segments[0];
    if (!head) return;
    const orb: Orb = { id: this.state.nextOrbId++, x: head.x + 60, y: head.y, color: c, scale: 1 };
    this.state.orbs.push(orb);
    this._doEat(orb, performance.now());
  }

  die(): void {
    this._triggerDeath(performance.now());
  }

  // ── player movement ─────────────────────────────────────────────────────────

  private _movePlayer(dt: number, now: number): void {
    const s = this.state;
    const head = s.segments[0];
    if (!head) return;

    const dx = s.pointer.x - head.x;
    const dy = s.pointer.y - head.y;
    const dLen = Math.sqrt(dx * dx + dy * dy);
    if (dLen < 2) return;

    const boosting = s.pointerDown && s.segments.length > BOOST_MIN_LENGTH;
    const speed = (boosting ? SPEED_BASE * SPEED_BOOST : SPEED_BASE) * (dt / 1000);
    const step = Math.min(speed, dLen);
    const nx = head.x + (dx / dLen) * step;
    const ny = head.y + (dy / dLen) * step;

    // shift segments: each segment follows the one in front
    for (let i = s.segments.length - 1; i > 0; i--) {
      const prev = s.segments[i - 1]!;
      const cur = s.segments[i]!;
      const fx = prev.x - cur.x, fy = prev.y - cur.y;
      const fd = Math.sqrt(fx * fx + fy * fy);
      if (fd > SEG_SPACING) {
        cur.x += (fx / fd) * (fd - SEG_SPACING);
        cur.y += (fy / fd) * (fd - SEG_SPACING);
      }
    }
    head.x = nx;
    head.y = ny;

    // wall collision
    if (head.x < SEG_R || head.x > ARENA_W - SEG_R ||
        head.y < SEG_R || head.y > ARENA_H - SEG_R) {
      this._triggerDeath(now);
      return;
    }

    // boost consume
    if (boosting) {
      s.boostConsumeAcc += BOOST_CONSUME_RATE * (dt / 1000);
      while (s.boostConsumeAcc >= 1 && s.segments.length > BOOST_MIN_LENGTH) {
        s.segments.pop();
        s.boostConsumeAcc -= 1;
      }
    } else {
      s.boostConsumeAcc = 0;
    }
  }

  // ── eating ──────────────────────────────────────────────────────────────────

  private _checkEat(now: number): void {
    const s = this.state;
    const head = s.segments[0];
    if (!head) return;
    for (let i = s.orbs.length - 1; i >= 0; i--) {
      const orb = s.orbs[i]!;
      if (dist(head, orb) < SEG_R + ORB_R) {
        this._doEat(orb, now);
        break; // one per tick
      }
    }
  }

  private _doEat(orb: Orb, now: number): void {
    const s = this.state;
    // remove orb
    s.orbs = s.orbs.filter(o => o.id !== orb.id);
    // append segment at tail
    const tail = s.segments[s.segments.length - 1]!;
    s.segments.push({ x: tail.x, y: tail.y, color: orb.color });
    // squash head
    s.headSquash = 1.4;
    s.headSquashDecay = (1.4 - 1) / 120; // back to 1 in 120ms
    // respawn orb
    s.orbs.push(spawnOrb(s.nextOrbId++, s.segments));
    // check for 3+ consecutive same-color runs
    this._scheduleConsecutivePops(now);
  }

  // ── pop detection ────────────────────────────────────────────────────────────

  private _scheduleConsecutivePops(now: number): void {
    const s = this.state;
    // Find all consecutive runs of length >= 3, not already pending
    const pendingSet = new Set(s.pendingPops.flatMap(p => p.indices));
    let i = 0;
    while (i < s.segments.length) {
      const color = s.segments[i]!.color;
      let j = i;
      while (j < s.segments.length && s.segments[j]!.color === color) j++;
      const runLen = j - i;
      if (runLen >= 3) {
        const indices = Array.from({ length: runLen }, (_, k) => i + k);
        // Only schedule if not already pending
        if (!indices.some(idx => pendingSet.has(idx))) {
          s.pendingPops.push({ indices, fireAt: now + POP_WARN_MS });
        }
      }
      i = j;
    }
  }

  private _fireReadyPops(now: number): void {
    const s = this.state;
    const ready = s.pendingPops.filter(p => now >= p.fireAt);
    if (ready.length === 0) return;

    s.pendingPops = s.pendingPops.filter(p => now < p.fireAt);

    // collect all indices to remove (may overlap if multiple pops)
    const toRemove = new Set<number>();
    for (const pop of ready) {
      for (const idx of pop.indices) toRemove.add(idx);
    }

    // score
    const combo = now < s.comboExpiry ? s.combo + 1 : 1;
    s.combo = combo;
    s.comboExpiry = now + COMBO_WINDOW_MS;
    const totalPopped = toRemove.size;
    s.score += SCORE_BASE * totalPopped * totalPopped * combo;

    // emit particles from popped positions
    for (const idx of toRemove) {
      const seg = s.segments[idx];
      if (seg) this._burst(seg.x, seg.y, seg.color, 8);
    }

    // remove segments (descending order to keep indices stable)
    const sorted = Array.from(toRemove).sort((a, b) => b - a);
    for (const idx of sorted) s.segments.splice(idx, 1);

    // screen shake for big pops
    if (totalPopped >= SHAKE_MIN_POP) {
      s.shakeUntil = now + SHAKE_MS;
    }

    // update best
    if (s.score > s.best) {
      s.best = s.score;
      save('best', s.best);
    }

    // re-check for newly formed runs after removal
    this._scheduleConsecutivePops(now);
  }

  // ── rivals ──────────────────────────────────────────────────────────────────

  private _moveRivals(dt: number, now: number): void {
    const s = this.state;
    for (const rival of s.rivals) {
      if (rival.dead) {
        if (now >= rival.respawnAt) {
          // respawn
          const fresh = makeRival(rival.id, now);
          rival.segments = fresh.segments;
          rival.targetOrb = null;
          rival.dead = false;
        }
        continue;
      }
      const head = rival.segments[0]!;

      // pick a target orb
      if (rival.targetOrb === null || !s.orbs.find(o => o.id === rival.targetOrb)) {
        let best: Orb | null = null, bestD = Infinity;
        for (const orb of s.orbs) {
          const d = dist(head, orb);
          if (d < bestD) { bestD = d; best = orb; }
        }
        rival.targetOrb = best ? best.id : null;
      }

      const targetOrb = rival.targetOrb !== null ? s.orbs.find(o => o.id === rival.targetOrb) : null;

      // steer: blend toward orb target, avoid walls
      let tx = head.x, ty = head.y;
      if (targetOrb) { tx = targetOrb.x; ty = targetOrb.y; }

      // wall avoidance
      const WALL_PAD = 150;
      if (head.x < WALL_PAD) tx += 200;
      if (head.x > ARENA_W - WALL_PAD) tx -= 200;
      if (head.y < WALL_PAD) ty += 200;
      if (head.y > ARENA_H - WALL_PAD) ty -= 200;

      const dx = tx - head.x, dy = ty - head.y;
      const d = Math.sqrt(dx * dx + dy * dy);
      if (d < 2) continue;
      const step = RIVAL_SPEED * (dt / 1000);
      head.x += (dx / d) * step;
      head.y += (dy / d) * step;

      // drag tail
      for (let i = rival.segments.length - 1; i > 0; i--) {
        const prev = rival.segments[i - 1]!;
        const cur = rival.segments[i]!;
        const fx = prev.x - cur.x, fy = prev.y - cur.y;
        const fd = Math.sqrt(fx * fx + fy * fy);
        if (fd > SEG_SPACING) {
          cur.x += (fx / fd) * (fd - SEG_SPACING);
          cur.y += (fy / fd) * (fd - SEG_SPACING);
        }
      }

      // eat orbs
      for (let i = s.orbs.length - 1; i >= 0; i--) {
        const orb = s.orbs[i]!;
        if (dist(head, orb) < SEG_R + ORB_R) {
          s.orbs.splice(i, 1);
          const tail = rival.segments[rival.segments.length - 1]!;
          rival.segments.push({ x: tail.x, y: tail.y, color: orb.color });
          s.orbs.push(spawnOrb(s.nextOrbId++));
          rival.targetOrb = null;
          break;
        }
      }
    }
  }

  private _checkRivalCollisions(now: number): void {
    const s = this.state;
    const head = s.segments[0]!;
    if (!head) return;

    for (const rival of s.rivals) {
      if (rival.dead) continue;
      const rHead = rival.segments[0]!;

      // rival head hits player body (skip head-to-head for simplicity)
      for (let i = 1; i < s.segments.length; i++) {
        const seg = s.segments[i]!;
        if (dist(rHead, seg) < SEG_R * 1.6) {
          // rival bursts into orbs
          this._burstRival(rival, now);
          break;
        }
      }

      if (rival.dead) continue;

      // player head hits rival body
      for (let i = 0; i < rival.segments.length; i++) {
        const seg = rival.segments[i]!;
        if (dist(head, seg) < SEG_R * 1.6) {
          this._triggerDeath(now);
          return;
        }
      }
    }
  }

  private _burstRival(rival: Rival, now: number): void {
    const s = this.state;
    // drop orbs at rival segment positions
    for (const seg of rival.segments) {
      s.orbs.push(spawnOrb(s.nextOrbId++, [seg]));
      this._burst(seg.x, seg.y, seg.color, 4);
    }
    // mark dead
    rival.dead = true;
    rival.respawnAt = now + RIVAL_RESPAWN_MS;
    rival.segments = [];
    rival.targetOrb = null;
  }

  // ── death ────────────────────────────────────────────────────────────────────

  private _triggerDeath(now: number): void {
    const s = this.state;
    s.phase = 'dead';
    // particles from all segments
    for (const seg of s.segments) {
      this._burst(seg.x, seg.y, seg.color, 5);
    }
    s.shakeUntil = now + SHAKE_MS * 2;
  }

  // ── particles ────────────────────────────────────────────────────────────────

  private _burst(x: number, y: number, color: number, count: number): void {
    const s = this.state;
    for (let i = 0; i < count; i++) {
      const angle = Math.random() * Math.PI * 2;
      const speed = rand(60, 200);
      s.particles.push({
        x, y,
        vx: Math.cos(angle) * speed,
        vy: Math.sin(angle) * speed,
        r: rand(3, 8),
        color,
        alpha: 1,
        life: 1,
        decay: rand(0.001, 0.003),
      });
    }
  }

  private _updateParticles(dt: number): void {
    const s = this.state;
    for (const p of s.particles) {
      p.x += p.vx * (dt / 1000);
      p.y += p.vy * (dt / 1000);
      p.vy += 120 * (dt / 1000); // gravity
      p.life -= p.decay * dt;
      p.alpha = Math.max(0, p.life);
    }
    s.particles = s.particles.filter(p => p.life > 0);
  }

  private _updateHeadSquash(dt: number): void {
    const s = this.state;
    if (s.headSquash !== 1) {
      s.headSquash = Math.max(1, s.headSquash - s.headSquashDecay * dt);
    }
  }
}
```

```typescript
import { SEG_R, SEG_COLORS, ORB_R, SHAKE_MS, ARENA_W, ARENA_H } from './constants';
import type { GameState, Segment } from './types';

export class Renderer {
  private canvas: HTMLCanvasElement;
  private ctx: CanvasRenderingContext2D;

  constructor(canvas: HTMLCanvasElement) {
    this.canvas = canvas;
    this.ctx = canvas.getContext('2d')!;
  }

  render(state: GameState, now: number): void {
    const { canvas, ctx } = this;
    const W = canvas.width;
    const H = canvas.height;
    const head = state.segments[0];
    if (!head) return;

    const camX = head.x - W / 2;
    const camY = head.y - H / 2;

    let shakeX = 0, shakeY = 0;
    if (now < state.shakeUntil && document.documentElement.dataset.motion !== 'off') {
      const mag = 3 * Math.max(0, (state.shakeUntil - now) / SHAKE_MS);
      shakeX = (Math.random() - 0.5) * 2 * mag;
      shakeY = (Math.random() - 0.5) * 2 * mag;
    }

    ctx.clearRect(0, 0, W, H);
    ctx.save();
    ctx.translate(-camX + shakeX, -camY + shakeY);

    this._drawArena(ctx);
    this._drawOrbs(ctx, state);
    this._drawRivals(ctx, state, now);
    this._drawSnake(ctx, state, now);
    this._drawParticles(ctx, state);
    this._drawVignette(ctx, camX, camY, W, H);

    ctx.restore();
  }

  private _drawArena(ctx: CanvasRenderingContext2D): void {
    ctx.fillStyle = '#0a0a1a';
    ctx.fillRect(0, 0, ARENA_W, ARENA_H);

    ctx.strokeStyle = 'rgba(241,239,255,0.04)';
    ctx.lineWidth = 1;
    const GRID = 80;
    for (let x = 0; x <= ARENA_W; x += GRID) {
      ctx.beginPath(); ctx.moveTo(x, 0); ctx.lineTo(x, ARENA_H); ctx.stroke();
    }
    for (let y = 0; y <= ARENA_H; y += GRID) {
      ctx.beginPath(); ctx.moveTo(0, y); ctx.lineTo(ARENA_W, y); ctx.stroke();
    }

    ctx.strokeStyle = 'rgba(0,245,160,0.4)';
    ctx.lineWidth = 3;
    ctx.strokeRect(2, 2, ARENA_W - 4, ARENA_H - 4);
  }

  private _drawOrbs(ctx: CanvasRenderingContext2D, state: GameState): void {
    for (const orb of state.orbs) {
      const [r, g, b] = SEG_COLORS[orb.color]!;
      ctx.save();
      ctx.translate(orb.x, orb.y);
      ctx.scale(orb.scale, orb.scale);

      const glow = ctx.createRadialGradient(0, 0, 0, 0, 0, ORB_R * 2.5);
      glow.addColorStop(0, `rgba(${r},${g},${b},0.4)`);
      glow.addColorStop(1, `rgba(${r},${g},${b},0)`);
      ctx.beginPath(); ctx.arc(0, 0, ORB_R * 2.5, 0, Math.PI * 2);
      ctx.fillStyle = glow; ctx.fill();

      const grad = ctx.createRadialGradient(-ORB_R * 0.3, -ORB_R * 0.3, 0, 0, 0, ORB_R);
      grad.addColorStop(0, `rgba(${Math.min(r+60,255)},${Math.min(g+60,255)},${Math.min(b+60,255)},1)`);
      grad.addColorStop(1, `rgb(${r},${g},${b})`);
      ctx.beginPath(); ctx.arc(0, 0, ORB_R, 0, Math.PI * 2);
      ctx.fillStyle = grad; ctx.fill();

      ctx.restore();
    }
  }

  private _drawRivals(ctx: CanvasRenderingContext2D, state: GameState, now: number): void {
    for (const rival of state.rivals) {
      if (rival.dead) continue;
      this._drawWorm(ctx, rival.segments, now, false, 1, undefined);
    }
  }

  private _drawSnake(ctx: CanvasRenderingContext2D, state: GameState, now: number): void {
    if (state.segments.length === 0) return;
    const pulsingSet = new Set<number>();
    for (const p of state.pendingPops) {
      for (const idx of p.indices) pulsingSet.add(idx);
    }
    this._drawWorm(ctx, state.segments, now, true, state.headSquash, pulsingSet);
  }

  private _drawWorm(
    ctx: CanvasRenderingContext2D,
    segments: Segment[],
    now: number,
    isPlayer: boolean,
    headSquash: number,
    pulsingSet: Set<number> | undefined,
  ): void {
    for (let i = segments.length - 1; i >= 0; i--) {
      const seg = segments[i]!;
      const [r, g, b] = SEG_COLORS[seg.color]!;
      const pulsing = pulsingSet?.has(i) ?? false;

      ctx.save();
      ctx.translate(seg.x, seg.y);

      const squash = (i === 0 && isPlayer) ? headSquash : 1;
      if (squash !== 1) ctx.scale(squash, 1 / squash);

      const radius = SEG_R * (i === 0 ? 1.15 : 1);

      if (pulsing) {
        const t = 0.5 + 0.5 * Math.sin(now * 0.025);
        ctx.shadowColor = `rgba(255,255,255,${t * 0.8})`;
        ctx.shadowBlur = 14;
      }

      const grad = ctx.createRadialGradient(-radius * 0.35, -radius * 0.35, 0, 0, 0, radius);
      grad.addColorStop(0, `rgba(${Math.min(r+80,255)},${Math.min(g+80,255)},${Math.min(b+80,255)},1)`);
      grad.addColorStop(0.6, `rgb(${r},${g},${b})`);
      grad.addColorStop(1, `rgba(${Math.max(r-60,0)},${Math.max(g-60,0)},${Math.max(b-60,0)},1)`);

      ctx.beginPath(); ctx.arc(0, 0, radius, 0, Math.PI * 2);
      ctx.fillStyle = grad; ctx.fill();

      // gloss
      ctx.beginPath();
      ctx.ellipse(-radius*0.25, -radius*0.3, radius*0.35, radius*0.2, -0.4, 0, Math.PI*2);
      ctx.fillStyle = 'rgba(255,255,255,0.35)'; ctx.fill();

      // outline
      ctx.beginPath(); ctx.arc(0, 0, radius, 0, Math.PI*2);
      ctx.strokeStyle = `rgba(${Math.max(r-40,0)},${Math.max(g-40,0)},${Math.max(b-40,0)},0.6)`;
      ctx.lineWidth = 1.5; ctx.stroke();

      // eyes on head
      if (i === 0 && isPlayer && segments.length > 1) {
        const next = segments[1]!;
        const dx = seg.x - next.x, dy = seg.y - next.y;
        const len = Math.sqrt(dx*dx + dy*dy) || 1;
        const fx = dx/len, fy = dy/len;
        const px = -fy, py = fx;
        const eyeR = radius * 0.28;
        const eyeOff = radius * 0.38;
        const eyeFwd = radius * 0.3;

        for (const side of [-1, 1] as const) {
          const ex = fx*eyeFwd + px*side*eyeOff;
          const ey = fy*eyeFwd + py*side*eyeOff;
          ctx.beginPath(); ctx.arc(ex, ey, eyeR, 0, Math.PI*2);
          ctx.fillStyle = '#fff'; ctx.fill();
          ctx.beginPath(); ctx.arc(ex+fx*eyeR*0.35, ey+fy*eyeR*0.35, eyeR*0.55, 0, Math.PI*2);
          ctx.fillStyle = '#1a1a2e'; ctx.fill();
          ctx.beginPath(); ctx.arc(ex-eyeR*0.2, ey-eyeR*0.2, eyeR*0.2, 0, Math.PI*2);
          ctx.fillStyle = 'rgba(255,255,255,0.8)'; ctx.fill();
        }
      }

      ctx.restore();
    }
  }

  private _drawParticles(ctx: CanvasRenderingContext2D, state: GameState): void {
    for (const p of state.particles) {
      const [r, g, b] = SEG_COLORS[p.color]!;
      ctx.save();
      ctx.globalAlpha = p.alpha;
      ctx.beginPath(); ctx.arc(p.x, p.y, p.r, 0, Math.PI*2);
      ctx.fillStyle = `rgb(${r},${g},${b})`; ctx.fill();
      ctx.restore();
    }
  }

  private _drawVignette(ctx: CanvasRenderingContext2D, camX: number, camY: number, W: number, H: number): void {
    const cx = camX + W/2, cy = camY + H/2;
    const grad = ctx.createRadialGradient(cx, cy, W*0.3, cx, cy, W*0.9);
    grad.addColorStop(0, 'rgba(0,0,0,0)');
    grad.addColorStop(1, 'rgba(0,0,0,0.45)');
    ctx.fillStyle = grad;
    ctx.fillRect(camX, camY, W, H);
  }
}
```

```typescript
import React, { useEffect, useRef, useState, useCallback } from 'react';
import { GameEngine, createInitialState } from '../game/engine';
import { Renderer } from '../game/renderer';
import { useSettings } from '../state/settings';
import { tap, pop as hapticPop, death as hapticDeath } from '../lib/haptics';
import { playEat, playPop, playDeath, playBest } from '../lib/audio';
import type { GameState } from '../game/types';
import HUD from './HUD';
import GameOver from './GameOver';
import SettingsDrawer from './SettingsDrawer';

export default function GameView() {
  const canvasRef = useRef<HTMLCanvasElement>(null);
  const engineRef = useRef<GameEngine | null>(null);
  const rendererRef = useRef<Renderer | null>(null);
  const rafRef = useRef<number>(0);
  const lastTimeRef = useRef<number>(0);
  const prevScoreRef = useRef<number>(0);
  const prevLengthRef = useRef<number>(0);
  const prevPhaseRef = useRef<string>('playing');

  const [hudState, setHudState] = useState<GameState | null>(null);
  const [showSettings, setShowSettings] = useState(false);
  const [isNewBest, setIsNewBest] = useState(false);

  const { sound, reducedMotion } = useSettings();

  const startGame = useCallback(() => {
    const canvas = canvasRef.current;
    if (!canvas) return;
    const state = createInitialState();
    const engine = new GameEngine(state);
    const renderer = new Renderer(canvas);
    engineRef.current = engine;
    rendererRef.current = renderer;
    prevScoreRef.current = 0;
    prevLengthRef.current = state.segments.length;
    prevPhaseRef.current = 'playing';
    setIsNewBest(false);
    setHudState({ ...state });
  }, []);

  // rAF loop
  useEffect(() => {
    if (!engineRef.current) return;

    const loop = (now: number) => {
      const engine = engineRef.current;
      const renderer = rendererRef.current;
      if (!engine || !renderer) return;

      const dt = Math.min(now - (lastTimeRef.current || now), 50); // cap at 50ms
      lastTimeRef.current = now;

      engine.tick(dt, now);
      renderer.render(engine.state, now);

      const s = engine.state;

      // juice side-effects
      if (s.score !== prevScoreRef.current) {
        const gained = s.score - prevScoreRef.current;
        if (gained > 0 && sound) playPop();
        if (gained > 0) hapticPop();
        if (s.score > s.best && prevScoreRef.current <= s.best) {
          setIsNewBest(true);
          if (sound) playBest();
        }
        prevScoreRef.current = s.score;
      }

      if (s.segments.length > prevLengthRef.current && sound) playEat();
      if (s.segments.length > prevLengthRef.current) tap();
      prevLengthRef.current = s.segments.length;

      if (s.phase === 'dead' && prevPhaseRef.current === 'playing') {
        if (sound) playDeath();
        hapticDeath();
      }
      prevPhaseRef.current = s.phase;

      setHudState({ ...s });
      if (s.phase === 'playing') {
        rafRef.current = requestAnimationFrame(loop);
      }
    };

    rafRef.current = requestAnimationFrame(loop);
    return () => cancelAnimationFrame(rafRef.current);
  }, [sound, hudState?.phase]); // re-subscribe when phase changes so dead screen doesn't keep looping

  // Resize canvas
  useEffect(() => {
    const canvas = canvasRef.current;
    if (!canvas) return;
    const resize = () => {
      canvas.width = window.innerWidth;
      canvas.height = window.innerHeight;
    };
    resize();
    window.addEventListener('resize', resize);
    return () => window.removeEventListener('resize', resize);
  }, []);

  // Start game on mount
  useEffect(() => {
    startGame();
  }, [startGame]);

  // Restart: stop old loop, start new
  const handleRestart = useCallback(() => {
    cancelAnimationFrame(rafRef.current);
    lastTimeRef.current = 0;
    startGame();
  }, [startGame]);

  // Pointer events → world coords
  const toWorld = useCallback((clientX: number, clientY: number): [number, number] => {
    const engine = engineRef.current;
    if (!engine) return [0, 0];
    const head = engine.state.segments[0];
    if (!head) return [0, 0];
    const canvas = canvasRef.current!;
    const W = canvas.width, H = canvas.height;
    const camX = head.x - W / 2;
    const camY = head.y - H / 2;
    return [clientX + camX, clientY + camY];
  }, []);

  useEffect(() => {
    const canvas = canvasRef.current;
    if (!canvas) return;

    const onMove = (e: PointerEvent) => {
      const [wx, wy] = toWorld(e.clientX, e.clientY);
      engineRef.current?.setPointer(wx, wy, e.buttons > 0 || e.pressure > 0);
    };
    const onDown = (e: PointerEvent) => {
      const [wx, wy] = toWorld(e.clientX, e.clientY);
      engineRef.current?.setPointer(wx, wy, true);
    };
    const onUp = (e: PointerEvent) => {
      const [wx, wy] = toWorld(e.clientX, e.clientY);
      engineRef.current?.setPointer(wx, wy, false);
    };

    canvas.addEventListener('pointermove', onMove);
    canvas.addEventListener('pointerdown', onDown);
    canvas.addEventListener('pointerup', onUp);
    canvas.addEventListener('pointercancel', onUp);
    return () => {
      canvas.removeEventListener('pointermove', onMove);
      canvas.removeEventListener('pointerdown', onDown);
      canvas.removeEventListener('pointerup', onUp);
      canvas.removeEventListener('pointercancel', onUp);
    };
  }, [toWorld]);

  // Pause on tab hidden
  useEffect(() => {
    const onVis = () => {
      if (document.hidden) {
        cancelAnimationFrame(rafRef.current);
      } else {
        lastTimeRef.current = 0; // reset dt to avoid big jump
        if (engineRef.current?.state.phase === 'playing') {
          rafRef.current = requestAnimationFrame(function loop(now) {
            const engine = engineRef.current;
            const renderer = rendererRef.current;
            if (!engine || !renderer) return;
            const dt = Math.min(now - (lastTimeRef.current || now), 50);
            lastTimeRef.current = now;
            engine.tick(dt, now);
            renderer.render(engine.state, now);
            setHudState({ ...engine.state });
            if (engine.state.phase === 'playing') {
              rafRef.current = requestAnimationFrame(loop);
            }
          });
        }
      }
    };
    document.addEventListener('visibilitychange', onVis);
    return () => document.removeEventListener('visibilitychange', onVis);
  }, []);

  // window.__snakePop testability hook
  useEffect(() => {
    (window as unknown as Record<string, unknown>)['__snakePop'] = {
      score: () => engineRef.current?.state.score ?? 0,
      length: () => engineRef.current?.state.segments.length ?? 0,
      die: () => engineRef.current?.die(),
      eatColor: (c: number) => engineRef.current?.eatColor(c),
    };
  }, []);

  return (
    <div style={{ position: 'relative', width: '100%', height: '100dvh', overflow: 'hidden' }}>
      <canvas
        ref={canvasRef}
        data-testid="game-canvas"
        style={{ display: 'block', width: '100%', height: '100%', touchAction: 'none' }}
      />

      {hudState && <HUD state={hudState} onSettings={() => setShowSettings(true)} />}

      {hudState?.phase === 'dead' && (
        <GameOver
          score={hudState.score}
          best={hudState.best}
          isNewBest={isNewBest}
          onRestart={handleRestart}
        />
      )}

      {showSettings && <SettingsDrawer onClose={() => setShowSettings(false)} />}
    </div>
```
